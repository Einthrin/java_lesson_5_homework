package eCommerceSimulation.business.concretes;

import eCommerceSimulation.business.abstracts.UserService;
import eCommerceSimulation.core.abstracts.InputVerificationService;
import eCommerceSimulation.core.abstracts.VerificationService;
import eCommerceSimulation.dataAccess.abstracts.UserDaoService;
import eCommerceSimulation.entities.concretes.User;

public class UserManager implements UserService
{
	
	InputVerificationService memberInformationVerification;
	UserDaoService userDaoService;
	VerificationService verificationService;
	
	public UserManager( InputVerificationService memberInformationVerification, UserDaoService userDaoService, VerificationService verificationService)
	{
		this.memberInformationVerification = memberInformationVerification;
		this.userDaoService = userDaoService;
		this.verificationService = verificationService;
	}

	@Override
	public void register(User user)
	{
		if(!memberInformationVerification.isVerification(user))
		{
			System.out.println("Kayıt işlemi başarısız. Kullanıcı bilgileri formatı yanluş.");
		}
		else if(!userDaoService.emailCheck(user))
		{
			System.out.println("Kayıt işlemi başarısız." + "\n" + "Bu email adresi daha önceden alınmış -->" + user.getEmail());
		}
		else
		{
			userDaoService.add(user);
			verificationService.sendVerificationMail();
		}
	}

	@Override
	public void login(String email, String password)
	{
		if(userDaoService.loginByCheck(email, password))
		{
			System.out.println("Giriş işlemi başarılı.");
		}
		else
		{
			System.out.println("Giriş işlemi başarısız." + "\n" + "Kullanıcı adı yanlış veya kullanıcı yok.");
		}
	}
	
}
